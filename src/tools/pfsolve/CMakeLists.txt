cmake_minimum_required(VERSION 3.18)
set(SOLUTION_NAME PFSolveSol)
project(${SOLUTION_NAME})
set(SOURCE_ROOT ${CMAKE_CURRENT_LIST_DIR})
if (WIN32)
	set(HOME_ROOT $ENV{USERPROFILE})
	string(REPLACE "\\" "/" HOME_ROOT "${HOME_ROOT}")
else()
	set(HOME_ROOT $ENV{HOME})
endif()
# natID SDK: env varijabla > git submodule (external/natID) > $HOME/natID.SDK
if (DEFINED ENV{NATID_SDK_ROOT})
	set(NATID_SDK_ROOT $ENV{NATID_SDK_ROOT})
elseif (EXISTS ${CMAKE_CURRENT_LIST_DIR}/../../../external/natID/natID.SDK/DevEnv/Common.cmake)
	get_filename_component(NATID_SDK_ROOT ${CMAKE_CURRENT_LIST_DIR}/../../../external/natID/natID.SDK ABSOLUTE)
else()
	set(NATID_SDK_ROOT ${HOME_ROOT}/natID.SDK)
endif()
message(STATUS "natID SDK: ${NATID_SDK_ROOT}")
include(${NATID_SDK_ROOT}/DevEnv/Common.cmake)
include(${NATID_SDK_ROOT}/DevEnv/MatrixLib.cmake)
include(${NATID_SDK_ROOT}/DevEnv/modSolver.cmake)
# artefakti u lokalni build folder (umjesto $HOME/natID.RAMDisk/Out)
foreach(CFG DEBUG RELEASE)
	set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_${CFG} ${CMAKE_CURRENT_BINARY_DIR}/out)
	set(CMAKE_LIBRARY_OUTPUT_DIRECTORY_${CFG} ${CMAKE_CURRENT_BINARY_DIR}/out)
	set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY_${CFG} ${CMAKE_CURRENT_BINARY_DIR}/out)
endforeach()
include(PFSolve.cmake)
