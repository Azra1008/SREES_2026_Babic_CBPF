\begin{frame}{Zadatak}
\begin{itemize}
  \item Proračun tokova snaga: za zadane injekcije snaga naći napone $V$ i uglove $\delta$.
  \item Metoda: nelinearni \textbf{strujni balans} ($\Delta I = 0$) u \textbf{polarnim} koordinatama.
  \item Nije samo rješavač --- pravi se \textbf{konvertor}:
  \begin{itemize}
    \item čita MATPOWER \texttt{.m} $\rightarrow$ generiše dTwin \texttt{.dmodl} NL model $\rightarrow$ dTwin ga riješi.
  \end{itemize}
  \item Zahtjevi: C++, konverzija u zasebnoj niti, real-time progres na drugoj niti, GUI, plugin (\texttt{dll}).
\end{itemize}
\end{frame}

\begin{frame}{Teorija --- strujni balans}
Proračunata struja: $I_i = \sum_k Y_{ik}V_k\,e^{j(\theta_{ik}+\delta_k)}$; \quad zadana: $I_i^{sp}=\frac{P_i-jQ_i}{V_i}e^{j\delta_i}$.

\vspace{2mm}
Jednačine (dvije po čvoru --- realni i imaginarni dio):
\begin{align*}
\sum_k Y_{ik}V_k\cos(\theta_{ik}+\delta_k) &= \frac{P_i\cos\delta_i + Q_i\sin\delta_i}{V_i}\\
\sum_k Y_{ik}V_k\sin(\theta_{ik}+\delta_k) &= \frac{P_i\sin\delta_i - Q_i\cos\delta_i}{V_i}
\end{align*}
\begin{itemize}
  \item \textbf{PQ}: nepoznate $\delta_i, V_i$. \quad \textbf{PV}: $\delta_i, Q_i^g$ ($V_i$ fiksan). \quad \textbf{Slack}: fiksno.
  \item Rješava se Newton--Raphsonom (rijedak Jakobijan).
\end{itemize}
\end{frame}
